#!/bin/bash
##########################################
#   Amazon Linux 2023                    #
#   Install httpd(Apache2 on Debian)     #
##########################################

# User data script to install httpd (Apache2 called on Debian) on Amazon Linux 2023

# Update system
sudo dnf update -y

# Install httpd called Apache2 for Debian
sudo dnf install -y httpd

# Enable and start httpd service
sudo systemctl enable httpd
sudo systemctl start httpd

# Test the installation
echo "<h1>Hello from Amazon Linux 2023 + Apache</h1>" > /var/www/html/index.html
